package javaapplication5;

import javax.swing.JOptionPane;

public class JavaApplication5 {
    
public static void main(String[] args) {
double distanciap,efi,litrosg;
        distanciap = Double.parseDouble(JOptionPane.showInputDialog("Digite a distância percorrida rm km "));
        litrosg = Double.parseDouble(JOptionPane.showInputDialog("Digite a quantidade de combustível gasto "));

       
        efi= calcular(distanciap, litrosg);

   
        JOptionPane.showMessageDialog(null, "O veículo percorre por litro" +efi);

        if (efi >= 15.0) {
            JOptionPane.showMessageDialog(null, "O veículo é econômico!");
        } else {
            JOptionPane.showMessageDialog(null, "O veículo não é econômico.");
        }
    }

    private static double calcular(double distancia, double litros) {
        return distancia / litros;
    }
}
